\newcommand{\bloco}[5]{
    % row, t, size, color, label
    \filldraw[
        fill=#4!30!white,
        draw=#4,
    ]
        (#2+1,#1) rectangle (#2+#3+1,#1+1)
        node at (#2 + 1.5,#1 + 0.5) {#5}
    ;
}
